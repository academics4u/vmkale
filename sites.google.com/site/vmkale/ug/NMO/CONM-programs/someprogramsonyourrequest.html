function n_r_method_IterationBased()
    clc   
    f = inline('3*(x)+sin(x)-exp(x)')
    fdash = inline('3+cos(x)-exp(x)')
    iterno=0;
    x1=input('Enter initial guess: ');
    TolValue=input('Enter TolValue: ');
    n=input('Number of Iterations: ');
    fprintf('\niterno\tx1\t\tf(x1)\t\tfdash(x1)\t\tx2');
    for i=1:n
        iterno=iterno+1;
        x2 = x1 - f(x1) / fdash(x1);
        fprintf('\n%d\t%f\t%f\t%f\t%f',iterno,x1,f(x1),fdash(x1),x2);
        x1 = x2;
    end
    fprintf('\nf(x2) = %e',f(x2))
    fprintf('\nIterations = %d',iterno-1)
    disp(f)
    fprintf('\nRoot of the above equation is (x2 = ) %5.7f\n',x2)
end

function GaussSeidel()
    clc
    a=input('Enter a (e.g. [20 2 6 28; 1 20 9 -23; 2 -7 -20 -57]): ')
    TolVal=input('TolVal (e.g. 0.000001) : ')
    n=size(a,1);
    x=zeros(1,n)
    iterno=1
    disp(x)
    iterate=1;
    while(iterate==1)
        for i=1 : n
            xold=x(i);
            sum=0;
             for j=1 : n
                 if j~=i
                     sum = sum + a(i,j)*x(j);
                 end
             end
             x(i)=(a(i,n+1)-sum)/a(i,i);
            if(abs(x(i)-xold)<TolVal)
                    iterate=0;
            end          
        end
        iterno=iterno+1;
        disp(iterno)
        disp(x)
    end
    x=x
end

function St_line()
    clc
    x=input('x (e.g. [1 2 3 4 5]): ')
    y=input('y (e.g. [28    37    50    67    88]): ')
    n=size(x,2)

    % Straight Line
    sX = 0; sY = 0; sX2 = 0; sYX = 0;
    for i = 1 : n 
       sX = sX + x(i);
       sY = sY + y(i);
       sX2 = sX2 + x(i) * x(i);
       sYX = sYX + y(i) * x(i);
    end
    D = (sX * sX - n * sX2 )
    a = ( sY * sX  -  n * sYX ) / D;
    b = ( sX * sYX  -  sY * sX2 ) / D;
    fprintf('\n\nThe equation is:  Y = %f X %+f\n',a, b)
end

function Parabola()
    clc
    x=input('x (e.g. [1 2 3 4 5]): ')
    y=input('y (e.g. [28    37    50    67    88]): ')
    n=size(x,2)

    sX = 0; sY = 0; sX2 = 0; sX3= 0;
    sX4 = 0; sYX = 0; sYX2 = 0;
    for i = 1 : n
       sX = sX + x(i);
       sY = sY + y(i);
       sX2 = sX2 + x(i) * x(i);
       sX3 = sX3 + x(i) * x(i) * x(i);
       sX4 = sX4 + x(i) * x(i) * x(i) * x(i);
       sYX = sYX + y(i) * x(i);
       sYX2 = sYX2 + y(i) * x(i) * x(i);
    end
    D=sX2*(sX2*sX2 -sX * sX3 ) - sX*(sX3*sX2  -  sX*sX4) + n * (sX3 * sX3  -  sX2*sX4)
    a =  sY * ( sX2*sX2  -  sX*sX3) -sX*(sYX*sX2  -  sX*sYX2) + n * (sYX*sX3  -  sX2*sYX2);
    a = a / D;
    b=sX2*(sYX*sX2  -  sX*sYX2)-sY * (sX3*sX2  -  sX*sX4)+n* (sX3 * sYX2  -  sYX*sX4);
    b = b / D;
    c=sX2*(sX2*sYX2  - sYX*sX3)-sX*(sX3*sYX2- sYX* sX4)+sY*(sX3*sX3  -  sX2*sX4);
    c = c / D;
    fprintf('\n\nThe equation is:  Y = %f X^2 %+f X %+f\n\n',a, b,c)
end

function PowerCurve()
    clc
    xx=input('x (e.g. [1 2 3 4 5]): ')
    yy=input('y (e.g. [28    37    50    67    88]): ')
    n=size(xx,2)

    % Power curve
    x=log(xx)
    y=log(yy)
 
    sX = 0; sY = 0; sX2 = 0; sYX = 0;
    for i = 1 : n  
       sX = sX + x(i);
       sY = sY + y(i);
       sX2 = sX2 + x(i) * x(i);
       sYX = sYX + y(i) * x(i);
    end
    D = (sX * sX - n * sX2 );
    a = ( sY * sX  -  n * sYX ) / D;
    b = ( sX * sYX  -  sY * sX2 ) / D;
   
    a1=exp(b)
    b1=a
    fprintf('\n\nThe Power equation is:  Y = %f X^(%f)\n',a1, b1)
end
